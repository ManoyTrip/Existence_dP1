/*input: 
    genListdP1
    groupListdP1
    orbitIntersectionsListdP2
e.g., after running the E7 and E8 file */

/* We produce a sequence "dP2Blowup1pt" where the i-th position contains the number (Urabe numbering) 
   of the type of dP1 obtained from blowing up a point in general position on a dP2 
   of type i (Trepalin numbering) */
dP2Blowup1pt := [];
dP1TypeBlowdown := [];

/* We construct the sequence "dP1TypeBlowdown", which for every type of dP1 contains a set of all possible types of dP2 obtained from blowing down a line defined over Fq*/
for i in [1..112] do
  gen := genListdP1[i];
  orbitsdP1 := Orbits(groupListdP1[i]);
  dP1TypeBlowdown[i] := {IntegerRing() | };
  /*For each line which is defined over the base field, we determine which type of dP2 is obtained upon contracting this line.*/
  for O in orbitsdP1 do
    if #O eq 1 then
      orbitsdP2 := contract_orbit(O, orbitsdP1, M1);
      multiplicitiesdP2 := intersection_pattern(gen, orbitsdP2, M1);
      typedP2 := Position(orbitIntersectionsListdP2, multiplicitiesdP2);
      Include(~(dP1TypeBlowdown[i]), typedP2);
    end if;
  end for;
end for;

/* We construct the sequence "dP2Blowup1pt", which consists for every type of dP2, the type of dP1 obtained by blowing up a sufficiently general rational point. In the process, we confirm that the obtained dP1 type is uniquely determined by the dP2 type.*/ 
uniqueBlowup := true;
for i in [1..60] do
  cnt := 0;
  for j in [1..#dP1TypeBlowdown] do
    if i in dP1TypeBlowdown[j] then
      cnt := cnt + 1;
      dP2Blowup1pt[i] := j;
    end if;
  end for;
  
  /*We check that every type of dP2 blows up to a unique type of dP1*/
  if cnt gt 1 then
    uniqueBlowup := false;
  end if;
end for;

uniqueBlowup;
