poly<q> := PolynomialRing(IntegerRing(), 1);
poly2<sqrt_q> := PolynomialRing(IntegerRing(), 1);
upperBoundPointsOnLinesDP2 := [Poly | ];
lowerBoundOddDP2 := [];
lowerBoundEvenDP2 := [];

for i in [1..60] do
  Append(~upperBoundPointsOnLinesDP2, 0);
  typesMultiplicity := orbitIntersectionsListdP2[i];
  types := MultisetToSet(typesMultiplicity);
  types := SetToSequence(types);
  
  sort_two_orbit_types := function(x, y)
   return sort_seq_by_multiplicity(x, y, typesMultiplicity);
  end function;

  Sort(~types, sort_two_orbit_types);

  if #types[1] eq 0 then
    upperBoundPointsOnLinesDP2[i] := upperBoundPointsOnLinesDP2[i] + Multiplicity(typesMultiplicity, types[1])*(q+1);
  end if;
  for type in types do
    if (#type ne 0) and (Minimum(type) ne 0) then
      upperBoundPointsOnLinesDP2[i] := upperBoundPointsOnLinesDP2[i] + Minimum(type)*Multiplicity(typesMultiplicity, type);
    end if;
  end for;
  
  q_odd := 3;
  while (Evaluate((q^2 + traceListdP2[i]*q + 1) - upperBoundPointsOnLinesDP2[i], [q_odd]) - Floor(q_odd + 1 + 6*Sqrt(q_odd))) le 0 do
    q_odd := q_odd + 2;
  end while;
  lowerBoundOddDP2[i] := q_odd;

  q_even := 2;
  while (Evaluate((q^2 + traceListdP2[i]*q + 1) - upperBoundPointsOnLinesDP2[i], [q_even]) - 2*(q_even + 1)) le 0 do
    q_even := q_even*2;
  end while;
  lowerBoundEvenDP2[i] := q_even;
end for;

for i in [1..60] do
  printf "%o. \n Max rational points on lines: %o \n", i, upperBoundPointsOnLinesDP2[i];
  printf "\ \ %o, %o. \n", lowerBoundOddDP2[i] , lowerBoundEvenDP2[i];
end for;
